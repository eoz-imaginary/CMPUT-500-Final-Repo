Inductive Z : Set :=
  | OZ : Z
  | pos : nat -> Z
  | neg : nat -> Z.
Definition IZ := pos 0.
Definition is_posn (x y : Z) :=
  match x, y with
  | pos n, pos m => n = m
  | _, _ => False
  end.
Proof.
intros.
congruence.
Qed.
unfold not in |- *.
intros.
cut (is_posn (pos n) (pos m)).
simpl in |- *.
exact H.
rewrite H0.
simpl in |- *.
reflexivity.
Qed.
Proof.
Admitted.
intros.
elim x.
left.
reflexivity.
intros.
right.
discriminate.
intros.
right.
discriminate.
Qed.
